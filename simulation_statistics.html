import statsmodels.api as sm

# Wybr zmiennych niezalenych (predyktorw) i zmiennej zalenej
X = df_only_relevant[COLUMNS_TO_ANALYZE]  # Zmienna niezalena
y = df_only_relevant['Served Customers']  # Zmienna zalena

# Dodanie staej do modelu (intercept)
X = sm.add_constant(X)

# Dopasowanie modelu regresji wielorakiej
model = sm.OLS(y, X).fit()

# Wywietlenie podsumowania wynikw
print(model.summary())
